add_executable(echoclient
EchoClient.cpp
./pb/echo_service.pb.cc)

if(WIN32)
  target_link_libraries(echoclient gayrpc ws2_32)
  if(DEFINED ENV{APPVEYOR_CI})
    target_link_libraries(echoclient protobuf::libprotobuf ${BRYNET_LIBRARY})
  endif()
elseif(UNIX)
  find_package(Threads REQUIRED)
  target_link_libraries(echoclient PRIVATE gayrpc pthread brynet ssl crypto protobuf)
endif()
